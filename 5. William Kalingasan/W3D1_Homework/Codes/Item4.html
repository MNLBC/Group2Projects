split() splits a string into an array of substrings
var a = "Hello World";
var b = a.split(" ")[0]; //Hello

charAt() returns the character of a specified index of the string
var a = "William";
var b = a.charAt(3); // l

indexOf() returns the index of a specified character/substring in the string. First found instance of said character/substring
var a = "Jana Paulyn Dimapilis";
var b = a.indexOf("a"); // 1

replace() searches for a regex or value in a string and replaces it with the given value
var a = "William Kalingasan";
var b = a.replace("ill","opp"); // Woppiam Kalingasan

substring() returns the a substring of a string using start and end indexes
var a = "San Beda College Alabang";
var b = a.substring(0,10); // San Beda C

startsWith() checks whether a string begins with a specific value
var a = "My name is Hello";
var b = a.startsWith("My"); // true
