1. split(): Splits a string into an array of substrings
var str = "How are you doing today?";
var res = str.split("");
Result: How,are,you,doing,today?
2. charAt(): Returns the character at the specified index (position)
var str = "HELLO WORLD";
var res = str.charAt(0);
Result: H
3. indexOf(): Returns the position of the first found occurrence of a specified value in a string (position)
var str = "Hello world, welcome to the universe.";
var n = str.indexOf("welcome");
Result: 13
4. replace(): Searches a string for a specified value, or a regular expression, and returns a new string where the specified values are replaced occurrence of a specified value in a string (position)
var str = "Visit Microsoft!";
var res = str.replace("Microsoft", "OOCL");
Result: Visit OOCL!
5. substring(): Extracts the characters from a string, between two specified indices
var str = "Hello world!";
var res = str.substring(1, 4);
Result: ell
6. startsWith(): Checks whether a string begins with specified characters
var str = "Hello world, welcome to the universe.";
var n = str.startsWith("Hello");
Result: true
